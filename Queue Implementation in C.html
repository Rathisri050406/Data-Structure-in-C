#include <stdio.h>
#include <stdlib.h>
#include <limits.h>

// Node structure
struct lnode {
    int data;
    struct lnode* next;
};

typedef struct lnode node;

// Queue structure
struct Queue {
    node* front;
    node* rear;
};

// Create a new queue
struct Queue* createQueue() {
    struct Queue* p = (struct Queue*)malloc(sizeof(struct Queue));
    p->front = NULL;
    p->rear = NULL;
    return p;
}

// Enqueue an element into the queue
void enqueue(struct Queue* q, int val) {
    node* newNode = (node*)malloc(sizeof(node));
    if (newNode == NULL) {
        printf("Out of memory\n");
        return;
    }

    newNode->data = val;
    newNode->next = NULL;

    if (q->rear == NULL)  // First element
        q->front = newNode;
    else
        q->rear->next = newNode;

    q->rear = newNode;
    printf("\nEnqueued %d", val);
}

// Dequeue an element from the queue
int dequeue(struct Queue* q) {
    if (q->front == NULL) {
        return INT_MIN;
    }

    node* del = q->front;
    q->front = q->front->next;

    if (q->front == NULL) // If last element removed
        q->rear = NULL;

    int temp = del->data;
    free(del);
    return temp;
}

int main() {
    struct Queue* q1 = createQueue();
    struct Queue* q2 = createQueue();
    struct Queue* q3 = createQueue();

    // Queue 1 operations
    enqueue(q1, 5);
